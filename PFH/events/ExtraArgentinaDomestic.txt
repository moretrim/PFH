## Campaign events ##

# Some notes:
#
# - vassals have reduced MTTH to spice up the substate life
# - countries at war have reduced frequency of internal events
# - some types of event or debate have a topic frequency
# - the political parties, press rights, and public meetings types of reforms affect event frequency
# - some events such as debates are intended to target states, in which case the frequency for those that are
#   implemented as province events is fuzzed down to a quarter reflecting a very rough average of 4 province per state
# - similarly the frequency of state-targeting country events scales proportionally to the overall number of states
# - state-targeting country events should take care to consistently open their `random_state` scopes (with identical
#   limits) at the same point (typically it's the very first thing) across their options for RNG consistency purposes
# - the AI is set to neglect militancy-lowering options in times of high infamy

            factor = 0.8

                liberal = 15

                liberal = 20

            factor = 0.8

        modifier = {
            factor = 1.44

                NOT = { political_parties = harassment }
                NOT = { political_parties = underground_parties }
            }
        }

        modifier = {
            factor = 1.44
            country = { NOT = { public_meetings = no_meeting } }
        }

        modifier = {
            factor = 1.44
            country = { NOT = { press_rights = state_press } }
        }
    }

    option = {
        name = "EVTOPTA252010" # We must organise. Hold a meeting of the Restorers.

        state_scope = {
            any_pop = {
                limit = {
                    OR = {
                        pop_majority_ideology = conservative
                        pop_majority_ideology = reactionary
                    }
                }

                dominant_issue = {
                    factor = 0.3
                    value = protectionism
                }
            }
        }

        ai_chance = {
            factor = 60
        }
    }

    option = {
        name = "EVTOPTB252010" # Take to the streets to revive the Federalist spirit.

        state_scope = {
            any_pop = {
                dominant_issue = {
                    factor = -0.1
                    value = free_trade
                }

                consciousness = -2
            }
        }

        ai_chance = {
            factor = 25

            modifier = {
                factor = 2
                state_scope = {
                    average_consciousness = 5
                }
            }

            modifier = {
                factor = 2
                state_scope = {
                    free_trade = 30
                }
            }
        }
    }

    option = {
        name = "EVTOPTC252010" # Fret not. The Mazorca will be paying a visit to the troublemakers.

        state_scope = {
            any_pop = {
                limit = {
                    pop_majority_ideology = liberal

            }

            }
        }

            factor = 15
